💡 How This Demo Works

// Initialize uploader
const uploader = new PanhaImagePreview('#upload-container', {
  maxFiles: 8,
  maxSize: 5 * 1024 * 1024
});

// Configure CRUD (connects to Laravel-style API)
uploader.configureCRUD({
  uploadUrl: '/api/images',      // Simulated endpoint
  deleteUrl: '/api/images',      // Simulated endpoint
  loadUrl: '/api/images',        // Simulated endpoint
  fieldName: 'images[]'
}).autoConfigureCSRF();

// Listen to events
uploader.on('ServerUploadSuccess', (data) => {
  console.log('Upload success:', data.response);
});

// Upload to server
await uploader.uploadToServer({
  onProgress: (percent) => {
    console.log(`${percent}% complete`);
  }
});
Note: This demo simulates server responses. In production, configure real API endpoints.
